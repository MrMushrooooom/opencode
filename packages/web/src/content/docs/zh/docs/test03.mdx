### 多语言代码块

```python title="advanced-test.py"
#!/usr/bin/env python3
"""
Advanced test script for translation workflow testing.
This script demonstrates complex Python features.
"""

import json
import yaml
from typing import Dict, List, Optional, Union
from dataclasses import dataclass
from pathlib import Path

@dataclass
class TranslationConfig:
    """Configuration class for translation settings."""
    source_lang: str
    target_lang: str
    model: str
    quality: str = "high"
    preserve_format: bool = True

    def to_dict(self) -> Dict[str, Union[str, bool]]:
        """Convert config to dictionary."""
        return {
            "source_lang": self.source_lang,
            "target_lang": self.target_lang,
            "model": self.model,
            "quality": self.quality,
            "preserve_format": self.preserve_format
        }

class TranslationWorkflow:
    """Main workflow class for handling translations."""

    def __init__(self, config: TranslationConfig):
        self.config = config
        self.processed_files: List[str] = []
        self.errors: List[str] = []

    def process_file(self, file_path: Path) -> bool:
        """Process a single file for translation."""
        try:
            if not file_path.exists():
                self.errors.append(f"File not found: {file_path}")
                return False

            # Simulate translation process
            self.processed_files.append(str(file_path))
            return True

        except Exception as e:
            self.errors.append(f"Error processing {file_path}: {e}")
            return False

    def get_stats(self) -> Dict[str, Union[int, List[str]]]:
        """Get processing statistics."""
        return {
            "processed_count": len(self.processed_files),
            "error_count": len(self.errors),
            "processed_files": self.processed_files,
            "errors": self.errors
        }

def main():
    """Main function to run the translation workflow."""
    config = TranslationConfig(
        source_lang="en",
        target_lang="zh",
        model="anthropic/claude-sonnet-4-20250514"
    )

    workflow = TranslationWorkflow(config)

    # Test file processing
    test_files = [
        Path("docs/test01.mdx"),
        Path("docs/test02.mdx"),
        Path("docs/test03.mdx")
    ]

    for file_path in test_files:
        workflow.process_file(file_path)

    # Print results
    stats = workflow.get_stats()
    print(json.dumps(stats, indent=2))

if __name__ == "__main__":
    main()
```

### 带复杂逻辑的 Shell 脚本

```bash title="complex-workflow.sh"
#!/bin/bash

# Advanced workflow testing script
# This script tests complex scenarios

set -euo pipefail

# Configuration
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
LOG_FILE="${SCRIPT_DIR}/workflow-test.log"
ERROR_LOG="${SCRIPT_DIR}/workflow-errors.log"

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# Logging functions
log_info() {
    echo -e "${BLUE}[INFO]${NC} $(date '+%Y-%m-%d %H:%M:%S') - $1" | tee -a "$LOG_FILE"
}

log_success() {
    echo -e "${GREEN}[SUCCESS]${NC} $(date '+%Y-%m-%d %H:%M:%S') - $1" | tee -a "$LOG_FILE"
}

log_warning() {
    echo -e "${YELLOW}[WARNING]${NC} $(date '+%Y-%m-%d %H:%M:%S') - $1" | tee -a "$LOG_FILE"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $(date '+%Y-%m-%d %H:%M:%S') - $1" | tee -a "$ERROR_LOG"
}

# Test functions
test_file_creation() {
    local test_dir="$1"
    local file_count="$2"

    log_info "Testing file creation in $test_dir"

    for i in $(seq 1 "$file_count"); do
        local test_file="${test_dir}/test-file-${i}.mdx"
        echo "# Test File $i" > "$test_file"
        echo "Content for test file $i" >> "$test_file"

        if [[ -f "$test_file" ]]; then
            log_success "Created test file: $test_file"
        else
            log_error "Failed to create test file: $test_file"
            return 1
        fi
    done
}

test_file_modification() {
    local test_dir="$1"

    log_info "Testing file modification in $test_dir"

    for file in "$test_dir"/*.mdx; do
        if [[ -f "$file" ]]; then
            echo "" >> "$file"
            echo "<!-- Modified at $(date) -->" >> "$file"
            log_success "Modified file: $file"
        fi
    done
}

test_file_deletion() {
    local test_dir="$1"
    local pattern="$2"

    log_info "Testing file deletion in $test_dir with pattern: $pattern"

    local deleted_count=0
    for file in "$test_dir"/$pattern; do
        if [[ -f "$file" ]]; then
            rm "$file"
            ((deleted_count++))
            log_success "Deleted file: $file"
        fi
    done

    log_info "Deleted $deleted_count files"
}

# Main execution
main() {
    log_info "Starting complex workflow testing"

    # Create test directory
    local test_dir="${SCRIPT_DIR}/test-docs"
    mkdir -p "$test_dir"

    # Test scenarios
    log_info "=== Scenario 1: File Creation ==="
    test_file_creation "$test_dir" 3

    log_info "=== Scenario 2: File Modification ==="
    test_file_modification "$test_dir"

    log_info "=== Scenario 3: File Deletion ==="
    test_file_deletion "$test_dir" "test-file-*.mdx"

    log_success "All tests completed successfully!"

    # Cleanup
    rm -rf "$test_dir"
    log_info "Test directory cleaned up"
}

# Error handling
trap 'log_error "Script interrupted"; exit 1' INT TERM

# Run main function
main "$@"
```
